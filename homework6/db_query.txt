/* CREATE DB */

CREATE DATABASE toy_store;

/* CREATE TABLE AND DEFINE COLUMN */

CREATE TABLE toys(
    id serial primary key,
    category_id integer,
    "name" varchar,
    price integer,
    description varchar,
    created_at timestamp
);
CREATE TABLE categories(
    id serial primary key,
    "name" varchar,
    created_at timestamp
);
